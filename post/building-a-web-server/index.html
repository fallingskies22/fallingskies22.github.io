.intel_syntax noprefix
.globl _start

.section .text

_start:
    mov rdi, 2      
    mov rsi, 1
    mov rdx, 0
    mov rax, 41     # SYS_socket(tcp)
    syscall

    mov rdi, 3
    lea rsi, sockaddr
    mov rdx, 16
    mov rax, 49     # SYS_bind
    syscall

    mov rdi, 3
    mov rsi, 0
    mov rax, 50     # SYS_listen
    syscall

    fork:
        mov rdi, 3
        lea rsi, [0]
        lea rdx, [0]
        mov rax, 43     # SYS_accept
        syscall
        mov r8, rax

    
        mov rax, 57
        syscall         # SYS_fork

        cmp rax, 0
        je begin

        mov rdi, r8
        mov rax, 3      #SYS_close
        syscall
        jmp fork

    begin:
        mov rdi, 3
        mov rax, 3      #SYS_close
        syscall

        mov rdi, r8
        lea rsi, readstr
        mov rdx, 512
        mov rax, 0      # SYS_read
        syscall
        mov r15, rax

        xor rcx,rcx
        lea rcx,readstr
        cmp byte ptr [rcx], 'G'
        je GET
        POST:
            # find file path
            xor rcx, rcx
            lea rcx, readstr
            add rcx, 5
            mov r9, rcx
            loop:
                cmp byte ptr [rcx], ' '         
                je end
                add rcx,1
                jmp loop
            end:
                mov byte ptr [rcx],0

            lea rdi,[r9]
            mov rsi,0x41       #O_WR|O_CREAT
            mov rax,2       #SYS_open
            mov rdx,0777
            syscall
            mov r12, rax

            # find file contents
            xor rcx, rcx
            lea rcx, readstr
            loop3:
                mov eax, [rcx]
                cmp eax, 168626701
                je end3
                add rcx,1
                jmp loop3
            end3:
            add rcx, 4

            mov rdi, r12
            mov rsi, rcx
            lea r10, readstr
            sub rcx, r10
            sub r15, rcx
            mov rdx, r15
            mov rax, 1      #SYS_write
            syscall

            mov rdi, r12
            mov rax, 3      #SYS_close
            syscall

            mov rdi, r8
            lea rsi, writestr
            mov rdx, 19     
            mov rax, 1      #SYS_write
            syscall

            jmp EXIT

        GET:
            # find file path
            xor rcx, rcx
            lea rcx, readstr
            add rcx, 4
            mov r9, rcx
            loop2:
            cmp byte ptr [rcx], ' '         
            je end2
            add rcx,1
            jmp loop2
            end2:
            mov byte ptr [rcx],0

            lea rdi,[r9]
            mov rsi,0       #O_RDONLY
            mov rax,2       #SYS_open
            syscall
            mov r12, rax

            mov rdi, r12
            lea rsi, filestr
            mov rdx, 512
            mov rax, 0      #SYS_read
            syscall
            mov r10, rax

            mov rdi, r12
            mov rax, 3      #SYS_close
            syscall

            mov rdi, r8
            lea rsi, writestr
            mov rdx, 19     
            mov rax, 1      #SYS_write
            syscall

            mov rdi, r8
            lea rsi, filestr
            mov rdx, r10     
            mov rax, 1      #SYS_write
            syscall

            jmp EXIT
    EXIT:
    mov rdi, 0
    mov rax, 60     # SYS_exit
    syscall

.section .data
sockaddr:
    .2byte 2
    .2byte 0x5000
    .4byte 0
    .8byte 0
writestr:
    .ascii "HTTP/1.0 200 OK\r\n\r\n"
readstr:
    .ascii "12"
filestr:
    .ascii "12"
